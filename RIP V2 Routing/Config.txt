! ------------------------------------------------------------------
! R0 configuration
! ------------------------------------------------------------------
enable
configure terminal
hostname R0
!
interface gig0/0
 description LAN-R0
 ip address 192.168.1.1 255.255.255.0
 no shutdown
!
interface gig0/1
 description R0-to-R1
 ip address 10.0.0.1 255.255.255.252
 no shutdown
!
router rip
 version 2
 network 192.168.1.0
 network 10.0.0.0
 no auto-summary
!
end
write memory
